\newcommand\ipin[1]{
    \textcolor{green}{
        In [#1]:
      }
}
\newcommand\ipout[1]{
    \textcolor{red}{
      Out[#1]:
    }
}
\newcommand\ipincont{
    \textcolor{green}{
      ~~~...:
    }
}

\begin{semiverbatim}

}{
\end{tabular}
\end{semiverbatim}
\end{beamercolorbox}
}

\newcommand\ipinwrap[3]{
    %first argument is slide from which this is displayed
    %second argument is ipython counter
    %third is the text content
    \uncover<#1->{\ipin{#2}  & $\texttt{#3}$ \\}
}

\newcommand\ipouwrap[3]{
    %see \ipinwrap
    \uncover<#1->{\ipout{#2}  & $\texttt{#3}$ \\}
}
